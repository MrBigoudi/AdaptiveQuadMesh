258 void processKeyboardInput(CameraMovement direction, float deltaTime){

315 void updateCameraVectors(){
316 // calculate the new Front vector
317 maths::Vector3 front;
318 front.x(cos(glm::radians(mYaw)) * cos(glm::radians(mPitch)));
319 front.y(sin(glm::radians(mPitch)));
320 front.z(sin(glm::radians(mYaw)) * cos(glm::radians(mPitch)));
321 mFront = front.normalize();
322 // also re-calculate the Right and Up vector
323 mRight = maths::Vector3::cross(mFront, mWorldUp).normalize(); // normalize the vectors, because their length gets closer to 0 the more you look up or down which results in slower movement.
324 mUp = maths::Vector3::cross(mRight, mFront).normalize();
325 }
